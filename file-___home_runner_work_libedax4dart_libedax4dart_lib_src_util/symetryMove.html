int symetryMove(final int x, final int symetry) {
  var x_ = x; // ignore: non_constant_identifier_names
  if (symetry & 1 != 0) {
    x_ = (x_ & _octal070) | (7 - (x_ & 7));
  }
  if (symetry & 2 != 0) {
    x_ = (_octal070 - (x_ & _octal070)) | (x_ & 7);
  }
  if (symetry & 4 != 0) {
    x_ = ((x_ & _octal070) >> 3) | ((x_ & 7) << 3);
  }
  return x_;
}
