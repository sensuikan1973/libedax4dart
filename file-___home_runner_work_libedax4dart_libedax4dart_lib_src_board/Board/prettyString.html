String prettyString(final int currentColor) {
  final pStone = currentColor == TurnColor.black
      ? ColorChar.black
      : ColorChar.white;
  final oStone = currentColor == TurnColor.black
      ? ColorChar.white
      : ColorChar.black;

  final buffer = StringBuffer()..writeln('  A B C D E F G H');
